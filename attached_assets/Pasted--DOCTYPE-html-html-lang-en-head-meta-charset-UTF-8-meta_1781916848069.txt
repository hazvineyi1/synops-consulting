<script>
(function(){
  const pastMeetings=[
    {week:1,title:"Kickoff",duration:"22:10",items:["Define course scope and outcomes","Assign module owners","Set weekly cadence"]},
    {week:2,title:"Build review",duration:"18:47",items:["Storyline asset walkthrough","D2L embedding checkpoints","Accessibility (508) pass plan"]},
    {week:3,title:"QA & compliance",duration:"25:03",items:["Syllabus–LMS alignment","ABA tracker update","Open issues and blockers"]}
  ];
  let current={week:4,items:[
    {text:"Review action items from Week 3",done:false},
    {text:"Walk through new agenda topics",done:false},
    {text:"Confirm owners and due dates",done:false},
    {text:"",done:false}
  ]};
  let selectedWeek=3;

  const app=document.getElementById('app');
  const drawer=document.getElementById('drawer');
  const resizer=document.getElementById('resizer');
  const weeksList=document.getElementById('weeksList');
  const refTitle=document.getElementById('refTitle'),refBody=document.getElementById('refBody');
  const itemsEl=document.getElementById('items');
  const recBar=document.getElementById('recBar'),recBtn=document.getElementById('recBtn'),recLabel=document.getElementById('recLabel');
  const timerEl=document.getElementById('timer'),endBtn=document.getElementById('endBtn'),footNote=document.getElementById('footNote');
  const addBtn=document.getElementById('addBtn'),weekTag=document.getElementById('weekTag'),agendaCount=document.getElementById('agendaCount');
  const dateSub=document.getElementById('dateSub'),toast=document.getElementById('toast');
  const scrim=document.getElementById('scrim');

  dateSub.textContent=new Date().toLocaleDateString(undefined,{weekday:'long',month:'long',day:'numeric',year:'numeric'});
  const checkSVG='<svg width="13" height="13" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="3" stroke-linecap="round" stroke-linejoin="round"><path d="M5 12l5 5L20 7"/></svg>';
  function esc(s){return s.replace(/[&<>"']/g,c=>({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c]));}

  /* ---- open / close drawer ---- */
  function openDrawer(){app.classList.remove('closed');document.getElementById('collapseBtn').focus();}
  function closeDrawer(){app.classList.add('closed');document.getElementById('openerBtn').focus();}
  document.getElementById('collapseBtn').addEventListener('click',closeDrawer);
  document.getElementById('openerBtn').addEventListener('click',openDrawer);
  scrim.addEventListener('click',closeDrawer);

  /* ---- drag to resize (adjust margins) ---- */
  const MIN=264,MAX=620,DEFAULT=344;
  let dragging=false;
  function startDrag(e){dragging=true;app.classList.add('resizing');try{resizer.setPointerCapture(e.pointerId);}catch(_){}}
  function moveDrag(e){
    if(!dragging)return;
    const left=drawer.getBoundingClientRect().left;
    let w=e.clientX-left;
    w=Math.max(MIN,Math.min(MAX,w));
    app.style.setProperty('--side-w',w+'px');
  }
  function endDrag(){if(dragging){dragging=false;app.classList.remove('resizing');}}
  resizer.addEventListener('pointerdown',startDrag);
  window.addEventListener('pointermove',moveDrag);
  window.addEventListener('pointerup',endDrag);
  resizer.addEventListener('dblclick',()=>app.style.setProperty('--side-w',DEFAULT+'px'));
  // keyboard resize
  resizer.addEventListener('keydown',e=>{
    const cur=parseInt(getComputedStyle(app).getPropertyValue('--side-w'))||DEFAULT;
    if(e.key==='ArrowLeft'){app.style.setProperty('--side-w',Math.max(MIN,cur-16)+'px');e.preventDefault();}
    if(e.key==='ArrowRight'){app.style.setProperty('--side-w',Math.min(MAX,cur+16)+'px');e.preventDefault();}
  });

  /* ---- render ---- */
  function renderWeeks(){
    weeksList.innerHTML='';
    pastMeetings.forEach(m=>{
      const b=document.createElement('button');
      b.className='wk-row'+(m.week===selectedWeek?' active':'');
      b.innerHTML=`<span class="check">${checkSVG}</span><span class="wk-name">Week ${m.week}</span><span class="wk-dur">${m.duration}</span>`;
      b.addEventListener('click',()=>{selectedWeek=m.week;renderWeeks();renderRef();if(window.innerWidth<=820)closeDrawer();});
      weeksList.appendChild(b);
    });
  }
  function renderRef(){
    const m=pastMeetings.find(x=>x.week===selectedWeek);
    if(!m){refTitle.textContent='Select a meeting';refBody.innerHTML='<p class="ref-empty">Pick a past week above to read its notes.</p>';return;}
    refTitle.textContent=`Week ${m.week} — ${m.title}`;
    refBody.innerHTML=`<div class="ref-meta">${checkSVG} Transcribed · ${m.duration}</div><ul>${m.items.map(t=>`<li>${esc(t)}</li>`).join('')}</ul>`;
  }
  function renderItems(){
    itemsEl.innerHTML='';
    current.items.forEach((it,idx)=>{
      const li=document.createElement('li');
      li.className='item'+(it.done?' done':'');
      li.innerHTML=`<button class="tick" aria-label="Mark done">${checkSVG}</button><input class="txt" value="${esc(it.text)}" placeholder="New agenda item…"><button class="del" aria-label="Delete item"><svg width="15" height="15" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round"><path d="M18 6L6 18M6 6l12 12"/></svg></button>`;
      li.querySelector('.tick').addEventListener('click',()=>{it.done=!it.done;renderItems();});
      li.querySelector('.txt').addEventListener('input',e=>{it.text=e.target.value;updateCount();});
      li.querySelector('.del').addEventListener('click',()=>{current.items.splice(idx,1);renderItems();});
      itemsEl.appendChild(li);
    });
    updateCount();
  }
  function updateCount(){
    const filled=current.items.filter(i=>i.text.trim()).length;
    const done=current.items.filter(i=>i.done&&i.text.trim()).length;
    agendaCount.textContent=filled?`${done}/${filled} covered`:'';
  }
  addBtn.addEventListener('click',()=>{current.items.push({text:'',done:false});renderItems();const ins=itemsEl.querySelectorAll('.txt');ins[ins.length-1].focus();});

  /* ---- recording ---- */
  let recording=false,seconds=0,ticker=null;
  function fmt(s){return String(Math.floor(s/60)).padStart(2,'0')+':'+String(s%60).padStart(2,'0');}
  recBtn.addEventListener('click',()=>{
    if(!recording){recording=true;recBar.classList.add('live');recBtn.classList.add('is-live');recLabel.textContent='Recording';endBtn.disabled=false;footNote.textContent='Recording in progress…';ticker=setInterval(()=>{seconds++;timerEl.textContent=fmt(seconds);},1000);}
    else{recording=false;recBar.classList.remove('live');recBtn.classList.remove('is-live');recLabel.textContent='Resume';footNote.textContent='Paused.';clearInterval(ticker);}
  });
  endBtn.addEventListener('click',()=>{
    clearInterval(ticker);recording=false;recBar.classList.remove('live');recBtn.classList.remove('is-live');
    const dur=fmt(seconds);
    const archived={week:current.week,title:"Transcribed",duration:dur,items:current.items.filter(i=>i.text.trim()).map(i=>i.text.trim())};
    if(!archived.items.length)archived.items=["(no agenda items recorded)"];
    pastMeetings.push(archived);
    selectedWeek=archived.week;renderWeeks();renderRef();
    showToast(`Week ${current.week} transcribed · ${dur}`);
    current={week:current.week+1,items:[{text:'',done:false}]};
    seconds=0;timerEl.textContent='00:00';weekTag.textContent=`Week ${current.week} · In progress`;
    recLabel.textContent='Record';endBtn.disabled=true;footNote.textContent='Start recording to enable transcription.';
    renderItems();
  });
  function showToast(msg){toast.innerHTML=checkSVG+' '+msg;toast.classList.add('show');setTimeout(()=>toast.classList.remove('show'),2600);}

  renderWeeks();renderRef();renderItems();
})();
</script>
